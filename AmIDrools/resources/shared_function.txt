//function that check if the fact is locked
function boolean getLock(String s, Wois wois){
	if(wois.getLock(s)==true){
		System.out.println( "Lamp LOCK!" );
		return true;
	}else{
		System.out.println( "Lamp UNLOCK" );}
	return false;
}

function boolean setLock(String s, Wois wois){
	if(wois.setLock(s)==true){
		return true;
	}
	return false;
}
